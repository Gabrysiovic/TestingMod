#Adds healthy man modifier and gives a few free techs
REPLACE:je_sick_man_main = {
	icon = "gfx/interface/icons/event_icons/event_skull.dds"
	group = je_group_tanzimat
	is_shown_in_lobby = {
		c:TUR = THIS
	}
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	on_complete = {
		remove_modifier = sick_man_of_europe
		add_modifier = {
			name = AA_healthy_man_of_europe
			years = 20
			is_decaying = yes
		}
		custom_tooltip = {
			text = ottos_tech
			add_technology_researched = atmospheric_engine
			add_technology_researched = railways
			add_technology_researched = mechanical_tools
			add_technology_researched = percussion_cap
			add_technology_researched = logistics
			add_technology_researched = central_banking
			add_technology_researched = nationalism
		}
		trigger_event = {
			id = sick_man.2
		}
	}
	timeout = 10950
	on_timeout = {
		remove_modifier = sick_man_of_europe
		add_modifier = {
			name = dead_man_of_europe
			days = very_long_modifier_time
			is_decaying = yes
		}
		trigger_event = {
			id = sick_man.3
		}
		if = {
			limit = {
				is_in_power_bloc = yes
				is_power_bloc_leader = yes
			}
			custom_tooltip = sick_man_of_europe_lose_power_bloc
		}
		custom_tooltip = {
			text = je_sick_man_main_failure_movements_tt
			set_variable = {
				name = failed_tanzimat
				years = 20
			}
		}
		if = {
			limit = {
				has_dlc_feature = ip3_content
			}
			custom_tooltip = {
				text = eastern_crisis_consequence_tt
				set_variable = failed_sick_man_main
			}
		}
		if = {
			#failure immediately triggers eastern_crisis
			limit = {
				c:TUR ?= this
				has_dlc_feature = ip3_content
				NOT = {
					has_global_variable = eastern_crisis_happened
				}
			}
			add_contextless_journal_entry = je_great_eastern_crisis
		}
		cu:albanian = {
			remove_tradition_trait = tradition_rumelian
		}
		cu:bosniak = {
			remove_tradition_trait = tradition_rumelian
		}
		cu:sephardic = {
			remove_tradition_trait = tradition_rumelian
		}
	}
	current_value = {
		value = var:sick_man_var
	}
	goal_add_value = {
		add = 4
	}
	progressbar = yes
	weight = 10000
	should_be_pinned_by_default = yes
}

#Lowers acceptance requirment to make Afro-american a primary culture easier
REPLACE:je_acw_equality = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	group = je_group_usa_american_civil_war
	is_shown_in_lobby = {
		c:USA = THIS
	}
	complete = {
		cultural_acceptance_base = {
			target = cu:afro_american
			value >= 40
		}
	}
	on_complete = {
		custom_tooltip = {
			text = acw_african_american_culture
			set_variable = acw_gain_african_american
		}
		custom_tooltip = {
			text = reconstruction_progress
			change_variable = {
				name = reconstruction_var
				add = 1
			}
		}
	}
	on_monthly_pulse = {
		random_events = {
			180 = 0
			20 = acw_followup_events.4			# Forty Acres and a Mule
			20 = acw_followup_events.5			# Supremacist Violence
			20 = acw_followup_events.6			# Redeemers
		}
	}
	timeout = 4380
	invalid = {
		NOT = {
			any_scope_state = {
				state_region = {
					is_homeland = cu:afro_american
				}
			}
		}
	}
	should_be_pinned_by_default = yes
}
