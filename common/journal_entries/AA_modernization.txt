je_spring_reforms_government = {
	group = je_group_internal_affairs
	is_shown_when_inactive = {
		OR = {
			c:JAP ?= THIS			#JAPAN
			c:BRZ ?= THIS			#BRAZIL
			c:TUR ?= THIS			#OTTOMANS
			c:PER ?= THIS			#PERSIA
			c:CLM ?= THIS			#COLOMBIA
			c:MEX ?= THIS			#MEXICO
			c:POR ?= THIS			#PORTUGAL
			c:SPA ?= THIS			#SPAIN
			c:GRE ?= THIS			#GREECE
			c:SIA ?= THIS			#SIAM
			c:PAN ?= THIS			#PUNJAB
			c:AFG ?= THIS			#AFGHANISTAN
			c:POL ?= THIS			#POLSKA
			c:DAI ?= THIS			#DAINAM
			c:SHW ?= THIS			#SHEWA
			c:ETH ?= THIS			#ETHIOPIA
			c:PBC ?= THIS			#PERU-BOLIVIA
			c:BOL ?= THIS			#BOLIVIA
			c:KOR ?= THIS			#KOREA
			c:EGY ?= THIS			#EGYPT
			c:ARG ?= THIS			#ARGENTINA
			c:CHL ?= THIS			#CHILE
			c:WAL ?= THIS			#WALLACHIA
			c:MOL ?= THIS			#MOLDAVIA
			c:ROM ?= THIS			#ROMANIA
			c:SOK ?= THIS			#SOKOTO
		}
	}
	possible = {
		always = yes
	}
	complete = {
		AND = {
			NOT = {
				has_law = law_type:law_autocracy
			}
			NOT = {
				has_law = law_type:law_state_religion
			}
			NOT = {
				has_law = law_type:law_hereditary_bureaucrats
			}
			has_law = law_type:law_slavery_banned
		}
	}
	on_complete = {
		add_modifier = {
			name = AA_modernize_government_modifier
			months = 128
		}
	}
	weight = 200
	should_be_pinned_by_default = yes
}
