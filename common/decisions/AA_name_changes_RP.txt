AA_restore_name = {
	is_shown = {
		OR = {
			has_variable = AA_indochina_var
			has_variable = AA_federation_var
		}
	}
	possible = {
		OR = {
			has_variable = AA_indochina_var
			has_variable = AA_federation_var
		}
	}
	when_taken = {
		remove_variable = AA_indochina_var
		remove_variable = AA_federation_var
	}
}

AA_indochina_form = {
	is_shown = {
		AND = {
			OR = {
				c:DAI ?= THIS
				c:SIA ?= THIS
			}
			hidden_trigger = {
				NOT = {
					has_variable = AA_indochina_var
				}
			}
			hidden_trigger = {
				NOT = {
					has_modifier = RP_restoration
				}
			}
		}
	}
	possible = {
		has_technology_researched = nationalism
		sr:region_indochina = {
			any_scope_state = {
				owner = ROOT
				count >= 8
			}
		}
	}
	when_taken = {
		set_variable = {
			name = AA_indochina_var
			value = yes
		}
		add_modifier = {
			name = RP_restoration
			days = long_modifier_time
		}
	}
	ai_chance = {
		value = 0
	}
}
