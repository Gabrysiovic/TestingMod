zw_trigger_westernized_headgear_character_politician = {
	# also used for republic ruler
	exists = scope:character
	scope:character ?= {
		# culture isn't western culture
		zw_trigger_non_western_culture = yes
		# country has westernized
		owner ?= { zw_trigger_country_is_westernized = yes }
		# is politician
		has_role = politician
		# is not ruler, heir, child
		is_ruler = no
		is_heir = no
		age >= 15
		# is from a westernized ig
		zw_trigger_ig_is_westernized = yes
	}
}

zw_trigger_westernized_headgear_character_politician_executive = {
	# also used for republic ruler
	exists = scope:character
	scope:character ?= {
		# culture isn't western culture
		zw_trigger_non_western_culture = yes
		# country has westernized
		owner ?= { zw_trigger_country_is_westernized = yes }
		# is ig_industrialists politician or executive
		OR = {
			AND = {
				has_role = politician
				is_interest_group_type = ig_industrialists
				# is not ruler, heir, child
				is_ruler = no
				is_heir = no
				age >= 15
			}
			has_role = executive
		}
		# is from a westernized ig
		zw_trigger_ig_is_westernized = yes
	}
}
