zw_change_army_uniform_button = {
	name = "zw_change_army_uniform_button"
	desc = "zw_change_army_uniform_button_desc"
	effect = {
		# set variable
		if = {
			limit = { has_variable = zw_var_special_clothes_general }
			if = {
				limit = { var:zw_var_special_clothes_general < 13 }
				change_variable = {
					name = zw_var_special_clothes_general
					add = 1
				}
			}
			else = {
				remove_variable = zw_var_special_clothes_general
			}
		}
		else = {
			set_variable = {
				name = zw_var_special_clothes_general
				value = 1
			}
		}
		# button description
		if = {

		}
		else_if = {
