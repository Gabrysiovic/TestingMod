our_side_penetration = {
	battle = {
		every_combat_unit = {
			limit = {
				owner = scope:character.owner
			}
			add = {
				value = flotilla_pen_value
				multiply = current_manpower
			}
		}
	}
	divide = {
		battle = {
			every_combat_unit = {
				limit = {
					owner = scope:character.owner
				}
				add = current_manpower
			}
		}
		min = 1
	}
}

formation_penetration = {
	every_combat_unit = {
		add = {
			add = flotilla_pen_value
			multiply = current_manpower
		}
	}
	if = {
		limit = {
			any_combat_unit = {
				count >= 1
				manpower >= 1
			}
		}
		divide = {
			every_combat_unit = {
				add = current_manpower
			}
		}
	}
	else = {
		value = 0
	}
}
