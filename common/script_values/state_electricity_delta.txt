
#root = state
state_electricity_delta = {
	value = sg:electricity.state_goods_delta

	if = {
		limit = {
			has_variable = substation_multiplier
			has_building = building_substation
			is_production_method_active = {
				building_type = building_substation
				production_method = pm_substation_down
			}
		}

		subtract = {
			value = var:substation_multiplier
			add = 1 # Von der PM
			multiply = substation_electricity_multiplier
		}
	}
	else_if = {
		limit = {
			has_variable = substation_multiplier
			has_building = building_substation
			is_production_method_active = {
				building_type = building_substation
				production_method = pm_substation_up
			}
		}

		add = {
			value = var:substation_multiplier
			add = 1 # Von der PM
			multiply = substation_electricity_multiplier
		}
	}
}
